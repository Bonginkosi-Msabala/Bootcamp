  const emailInput = document.getElementById("email")
          const messageInput = document.getElementById("message")

          const formDetails = document.getElementById("my-form")

          function handleSubmit(event) {
            event.preventDefault()
      
          // console.log(emailInput);
          const isEmailValid = emailInput.value.trim() !== '' && emailInput.validity.valid
          console.log(isEmailValid);

          let isMessageValid = messageInput.value.trim() !== '';
          console.log({ isMessageValid });
      
          const isFormValid = isEmailValid && isMessageValid
      
          if(isFormValid) {

              // collect data from the form
              const formData = new FormData(event.target)
              console.log(formData);
              fetch('https://formspree.io/f/mrgnrkzz', 
                  {
                      method: 'POST',
                      body: formData,
                      headers: {
                          'Accept': 'application/json'
                      }
                  }
              )
              .then( response => response.json())
              .then( data => {
                  console.log(data);
                  if(data.ok) {
                      alert('Thank you email sent')
                  }
              })
      
             
          } else {
              // alert the user that the form is invalid
              if(isEmailValid !== true) {
                  // display a caution for email 
                  const emailSpan = document.getElementById('required-email')
                  console.log(emailSpan.classList);
                  emailSpan.classList.remove('emailrequest')
                  console.log(emailSpan.classList);
              }
              if(isMessageValid !== true) {
                  // display a caution message
                  const messageSpan = document.getElementById('required-message')
                  messageSpan.classList.remove('messagerequest')
              }
          }
      }
